-- 企业基本信息表
DROP TABLE IF EXISTS enterprise_basic_info;
CREATE TABLE enterprise_basic_info (
    id INT AUTO_INCREMENT PRIMARY KEY COMMENT '唯一标识符',
    company_name VARCHAR(255) NOT NULL COMMENT '企业名称',
    unified_social_credit_code VARCHAR(50) UNIQUE COMMENT '统一社会信用代码',
    establishment_date DATE COMMENT '成立日期',
    legal_representative VARCHAR(100) COMMENT '法定代表人',
    industry_category VARCHAR(100) COMMENT '行业大类',
    industry_subcategory VARCHAR(100) COMMENT '行业小类',
    enterprise_scale VARCHAR(50) COMMENT '国标企业规模',
    actual_controller TEXT COMMENT '实际控制人',
    total_assets TEXT COMMENT '总资产',
    net_assets TEXT COMMENT '净资产',
    annual_revenue TEXT COMMENT '年收入',
    listing_flag VARCHAR(10) COMMENT '上市标志',
    listing_date DATE COMMENT '上市时间',
    listing_location VARCHAR(100) COMMENT '上市地点',
    stock_code VARCHAR(50) COMMENT '股票代码',
    registered_capital_currency VARCHAR(20) COMMENT '注册资本币种',
    registered_capital DECIMAL(20,2) COMMENT '注册资本(万元)',
    paidin_capital_currency VARCHAR(20) COMMENT '实收资本币种',
    paidin_capital DECIMAL(20,2) COMMENT '实收资本(万元)',
    is_gov_financing_platform TEXT COMMENT '是否政府融资平台',
    registered_address TEXT COMMENT '注册地址',
    office_address TEXT COMMENT '办公地址',
    contact_phone VARCHAR(100) COMMENT '联系电话',
    finance_contact TEXT COMMENT '财务部联系方式',
    basic_account_bank TEXT COMMENT '基本户开户行',
    main_settlement_bank TEXT COMMENT '主要结算行',
    internal_rating VARCHAR(50) COMMENT '内部评级',
    external_rating VARCHAR(50) COMMENT '外部评级',
    account_opening_date DATE COMMENT '开户时间',
    is_planning_listing TEXT COMMENT '是否拟上市',
    is_group_client TEXT COMMENT '是否集团客户',
    affiliated_group TEXT COMMENT '所属集团',
    business_scope LONGTEXT COMMENT '营业范围',
    created_at TIMESTAMP DEFAULT CURRENT_TIMESTAMP COMMENT '创建时间'
) ENGINE=InnoDB DEFAULT CHARSET=utf8mb4 COLLATE=utf8mb4_unicode_ci COMMENT='企业基础信息表';

-- 股权结构信息表
DROP TABLE IF EXISTS equity_structure;
CREATE TABLE equity_structure (
    id INT AUTO_INCREMENT PRIMARY KEY COMMENT '主键ID',
    company_name VARCHAR(200) NOT NULL COMMENT '企业名称',
    credit_code VARCHAR(50) NOT NULL COMMENT '统一社会信用代码',
    shareholder_name VARCHAR(200) NOT NULL COMMENT '股东名称',
    contribution_method VARCHAR(100) COMMENT '出资方式',
    currency VARCHAR(20) COMMENT '出资币种',
    subscribed_amount DECIMAL(20,2) COMMENT '应投金额（万元）',
    paid_in_amount DECIMAL(20,2) COMMENT '实投金额（万元）',
    equity_ratio VARCHAR(20) COMMENT '股权比例（%）',
    investment_date DATE COMMENT '投资日期'
) ENGINE=InnoDB DEFAULT CHARSET=utf8mb4 COMMENT='股权结构信息表';
