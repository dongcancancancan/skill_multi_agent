DROP TABLE IF EXISTS policy_tools;


CREATE TABLE `policy_tools` (
    -- 政策工具名称（主键）
    `policy_tool` VARCHAR(100) NOT NULL COMMENT '政策工具名称，如"碳减排支持工具"',
    
    -- 工具性质描述
    `tool_property` VARCHAR(200) NOT NULL COMMENT '工具类型和性质，如"央行再贷款工具"',
    
    -- 工具额度（可为空）
    `tool_quota` VARCHAR(50) DEFAULT NULL COMMENT '政策工具的资金额度，如"100亿元"',
    
    -- 支持的市场主体范围
    `supported_market_entities` TEXT NOT NULL COMMENT '支持的企业类型，如"大中小微企业"',
    
    -- 资金投向领域
    `investment_field` TEXT NOT NULL COMMENT '资金使用领域，如"碳减排三大重点领域"',
    
    -- 贴息安排详情
    `discount_arrangement` TEXT NOT NULL COMMENT '贴息政策的具体安排',
    
    -- 符合政策的标准要素（可为空）
    `policy_standard_elements` TEXT DEFAULT NULL COMMENT '申请需满足的条件要求',
    
    -- 政策亮点（可为空）
    `highlights` TEXT DEFAULT NULL COMMENT '该政策的特殊优势或创新点',
    
    -- 业务操作要点
    `business_points` TEXT NOT NULL COMMENT '业务办理的具体要求和流程',
    
    -- 支持对象范围
    `support_objects` TEXT NOT NULL COMMENT '可申请该政策的金融机构类型',
    
    PRIMARY KEY (`policy_tool`)
) ENGINE=InnoDB DEFAULT CHARSET=utf8mb4 COLLATE=utf8mb4_unicode_ci COMMENT='政策工具信息表';
